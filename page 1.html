1. Opening a File


    
            file = open('example.txt', 'r')

2. Reading from a File using read()


    
            file = open('example.txt', 'r')
            content = file.read()
            print(content)
            file.close()

3. Reading from a File using readline()


    
            file = open('example.txt', 'r')
            line = file.readline()
            while line:
                print(line)
                line = file.readline()
            file.close()

4. Reading from a File using readlines()


    
            file = open('example.txt', 'r')
            lines = file.readlines()
            for line in lines:
                print(line)
            file.close()

5. Writing to a File


    
            file = open('example.txt', 'w')
            file.write('Hello, World!')
            file.close()

6. Appending to a File


    
            file = open('example.txt', 'a')
            file.write('Hello, World!')
            file.close()

7. Closing a File using close()


    
            file = open('example.txt', 'r')
            content = file.read()
            print(content)
            file.close()

8. Context Managers using with statement


    
            with open('example.txt', 'r') as file:
                content = file.read()
                print(content)
